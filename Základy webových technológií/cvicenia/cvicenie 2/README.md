```

    <h1>Moja super-truper stránka</h1>
  </header>
  <section id="main">
    <!-- page content -->
  </section>
  <section id="secondary">
    <!-- secondary sidebar - related content -->
  </section>
  <footer>
    <!-- footer content -->
  </footer>
</div>
```
